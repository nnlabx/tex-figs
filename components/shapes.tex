% CUSTOM SHAPE LIBRARY
% Defines reusable TikZ node shapes beyond the defaults.
%
% REQUIRES
%   \usetikzlibrary{shapes.geometric}
%   \usetikzlibrary{shapes.symbols}
%   \usetikzlibrary{shapes.arrows}
%   \usetikzlibrary{shapes.multipart}
%   \usetikzlibrary{fit}
%   \usetikzlibrary{calc}
%
% USAGE
%   \node[shape name] at (x,y) {label};
%   \node[shape name, fill=seafoam, text=black] at (x,y) {label};
%   Combine freely with node styles from nodes.tex.
%
% SHAPE INDEX
%   § 1  Base shape aliases       (rounded, sharp, pill, soft, hex, diamond)
%   § 2  Geometric primitives     (circle, ellipse, triangle, trapezoid)
%   § 3  Document / UI shapes     (folder, document, database, cloud)
%   § 4  Flow / process shapes    (parallelogram, chevron, arrow box)
%   § 5  Annotation shapes        (callout, flag, badge, tab)
%   § 6  Technical shapes         (component, actor, package, note corner)
%   § 7  Compound style aliases   (ready-made shape+color combos)
% ══════════════════════════════════════════════════════

% ──────────────────────────────────────────────────────
% SHARED GEOMETRY DEFAULTS
% ──────────────────────────────────────────────────────
\tikzset{
	% Inner padding applied to every shape unless overridden
	shape base/.style={
		inner sep    = 6pt,
		outer sep    = 0pt,
		text centered,
		font         = \normalsize\sffamily,
	},
}
% ══════════════════════════════════════════════════════
% § 1  BASE SHAPE ALIASES
%      These mirror the aliases used in nodes.tex so that
%      `shape` and `node style` keys compose cleanly.
% ══════════════════════════════════════════════════════
\tikzset{
	% Rounded rectangle (default diagram node)
	rounded shape/.style={
		shape base,
		shape           = rounded rectangle,
		rounded rectangle arc length = 180,
	},
	% Sharp rectangle
	sharp shape/.style={
		shape base,
		shape           = rectangle,
	},
	% Pill / stadium (fully rounded ends)
	pill shape/.style={
		shape base,
		shape           = stadium,
	},
	% Softly rounded rectangle (less aggressive than rounded shape)
	soft shape/.style={
		shape base,
		shape           = rectangle,
		rounded corners = 4pt,
	},
	% Regular hexagon
	hex shape/.style={
		shape base,
		shape           = regular polygon,
		regular polygon sides = 6,
	},
	% Diamond / rhombus
	diamond shape/.style={
		shape base,
		shape           = diamond,
		aspect          = 2,
	},
}
